  type: "POST"
  url: "http://127.0.0.1:3000/api/v1/user/login"
  dataType: "json"
  data:
    email: email
    password: password
).done((data) ->
  if data.error is `undefined`
    sessionStorage.setItem "cookie", data.cookie
